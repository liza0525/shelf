FORK / EXEC / WAIT

프로세스를 복제하고 교체하기

fork()는 현재 프로세스를 통째로 복제하고, execvp()는 그 복제본을 다른 프로그램으로 바꿔치기합니다. Python의 subprocess.run()이 내부적으로 하는 일이 바로 이것입니다.

#include <sys/wait.h> #include <unistd.h> #include <stdio.h> int main(void) { pid_t pid = fork(); // 프로세스를 복제한다 if (pid < 0) { perror("fork"); return 1; } else if (pid == 0) { // 자식 프로세스: exec 계열로 새 프로그램으로 "교체"됨 char *argv[] = {"/bin/ls", "-l", NULL}; execvp(argv[0], argv); perror("execvp"); // exec가 실패했을 때만 이 아래로 내려온다 _exit(127); } else { // 부모 프로세스: 자식이 끝날 때까지 대기 int status; waitpid(pid, &status, 0); printf("자식 종료, exit code = %d\n", WEXITSTATUS(status)); } return 0; }

BUILTIN VS EXTERNAL

cd는 왜 fork하면 안 될까

fork로 만든 자식이 chdir()을 호출해도 변경은 자식에게만 적용되고, 부모(셸) 프로세스의 작업 디렉터리는 그대로입니다. 그래서 cd/exit는 fork 없이 셸이 직접 처리해야 합니다.

#include <string.h> #include <stdlib.h> #include <unistd.h> #include <sys/wait.h> int run_command(char **argv) { if (argv[0] == NULL) return 0; // 빈 입력 // 빌트인 명령은 fork 없이 셸 프로세스 안에서 직접 처리 if (strcmp(argv[0], "cd") == 0) { const char *dir = argv[1] ? argv[1] : getenv("HOME"); if (chdir(dir) != 0) perror("cd"); return 0; } if (strcmp(argv[0], "exit") == 0) { exit(argv[1] ? atoi(argv[1]) : 0); } // 그 외에는 fork + exec으로 외부 프로그램 실행 pid_t pid = fork(); if (pid == 0) { execvp(argv[0], argv); perror("execvp"); _exit(127); } int status; waitpid(pid, &status, 0); return WEXITSTATUS(status); }

PIPE

| 하나로 두 프로세스 잇기

pipe()는 커널에 읽기/쓰기 fd 쌍을 만들고, dup2()로 자식의 표준입출력을 그 fd로 바꿔치기하면 두 프로그램이 셸을 거치지 않고 직접 데이터를 주고받습니다.

#include <unistd.h> #include <sys/wait.h> // "ls | grep foo" 재현: 파이프 하나로 두 명령을 연결 int fds[2]; pipe(fds); // fds[0] = 읽기 끝, fds[1] = 쓰기 끝 pid_t p1 = fork(); if (p1 == 0) { // 왼쪽 명령: ls dup2(fds[1], STDOUT_FILENO); // stdout을 파이프 쓰기 끝으로 교체 close(fds[0]); close(fds[1]); execlp("ls", "ls", NULL); _exit(127); } pid_t p2 = fork(); if (p2 == 0) { // 오른쪽 명령: grep foo dup2(fds[0], STDIN_FILENO); // stdin을 파이프 읽기 끝으로 교체 close(fds[0]); close(fds[1]); execlp("grep", "grep", "foo", NULL); _exit(127); } close(fds[0]); // 부모는 파이프 양쪽 다 필요 없음 close(fds[1]); waitpid(p1, NULL, 0); waitpid(p2, NULL, 0);

REDIRECTION

>와 <도 결국 dup2

파일 리다이렉션은 파이프와 원리가 같습니다. open()으로 연 파일 디스크립터를 dup2()로 표준입출력 번호(0, 1) 자리에 밀어넣으면 됩니다.

#include <fcntl.h> #include <unistd.h> // "cmd > out.txt" : 표준출력을 파일로 리다이렉션 int fd = open("out.txt", O_WRONLY | O_CREAT | O_TRUNC, 0644); dup2(fd, STDOUT_FILENO); // STDOUT(1)이 이제 out.txt를 가리킨다 close(fd); // 원래 fd 번호는 더 이상 필요 없음 // "cmd < in.txt" : 표준입력을 파일에서 읽기 int fd_in = open("in.txt", O_RDONLY); dup2(fd_in, STDIN_FILENO); close(fd_in); execvp(argv[0], argv); // exec 이후에도 리다이렉션은 그대로 유지된다

PARSING

문자열을 토큰 배열로

5주차의 strtok과 3주차의 realloc 기반 동적 배열을 그대로 재사용해, 입력 한 줄을 execvp가 요구하는 char *argv[] 형태로 바꿉니다.

#include <string.h> #include <stdlib.h> // "ls -l /tmp" -> argv[] = {"ls", "-l", "/tmp", NULL} char **tokenize(char *line) { int capacity = 8, count = 0; char **argv = malloc(sizeof(char *) * capacity); // 3주차: 동적 배열 char *tok = strtok(line, " \t\n"); // 5주차: strtok으로 공백 분리 while (tok != NULL) { if (count >= capacity - 1) { capacity *= 2; argv = realloc(argv, sizeof(char *) * capacity); } argv[count++] = tok; // strtok은 line 버퍼를 직접 잘라 반환 tok = strtok(NULL, " \t\n"); } argv[count] = NULL; // execvp가 요구하는 NULL 종단 return argv; }

주의: argv의 각 원소는 line 버퍼 안을 가리키므로, line보다 먼저 free하면 안 됨

확인해보기

스스로 점검하기

1

fork() 반환값이 0이면 어느 쪽 프로세스인가요?

자식 프로세스입니다. 부모에게는 자식의 pid가 반환됩니다

2

dup2(fds[1], STDOUT_FILENO) 직후에도 close(fds[1])가 필요한 이유는?

dup2는 fd를 복제할 뿐 원본은 그대로 열려 있어, 닫지 않으면 읽는 쪽이 EOF를 받지 못함

3

execvp가 실패하면 자식 프로세스는 어떻게 되나요?

exec 실패 후에도 코드가 계속 실행되므로, perror 후 _exit()로 반드시 자식을 종료시켜야 함

PITFALLS

흔히 저지르는 실수

01
fork 후 자식/부모 fd 안 닫기 자식에서 파이프 읽기/쓰기 끝을 닫지 않으면 상대편이 EOF를 못 받아 grep이 무한 대기
02
dup2 순서 실수 dup2로 표준입출력을 바꾸기 전에 원본 fd를 먼저 close해버리면 리다이렉션이 깨짐; close는 dup2 다음에
03
좀비 프로세스 파이프에 자식이 여러 개인데 일부만 waitpid하면 나머지는 defunct 상태로 남음; 모든 자식 pid를 wait
04
exec 실패 처리 누락 execvp 실패 후 _exit 없이 넘어가면 자식이 부모 셸 코드를 그대로 이어서 실행해버림
